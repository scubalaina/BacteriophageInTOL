abbreviation = description
RNAP_unrooted = concatenated B and B' sequences aligned between taxa
noiso = excludes cultured bacteriophage sequences
withiso = includes cultured bacteriophage sequences
gt30 = input alignment trimmed where at least 30% of the sequences did not have gaps
gt50 = input alignment trimmed where at least 50% of the sequences did not have gaps
gt70 = input alignment trimmed where at least 70% of the sequences did not have gaps
95sites = input alignment with the fastest 5% of evolving sites were removed
.fa = alignment file
TIGER_output = output file of TIGER anaylsis
