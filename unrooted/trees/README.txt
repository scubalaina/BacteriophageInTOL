abbreviation = description
RNAP_unrooted = concatenated B and B' sequences aligned between taxa
noiso = excludes cultured bacteriophage sequences
withiso = includes cultured bacteriophage sequences
gt30 = input alignment trimmed where at least 30% of the sequences did not have gaps
95sites = input alignment with the fastest 5% of evolving sites were removed
.nwk = Newick formatted treefile
.ufboot = ultrafast bootstrap replicates file
IC1.nwk = relative internode certainty values as branch support values in newick tree
IC2.nwk = absolute internode certainty values as branch support values in newick tree
ufboot.nwk = ultrafast bootstrap values based on 1000 replicates reported as a percent out of 100 as branch support values in newick tree
